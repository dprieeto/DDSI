--> Procedimiento en oracle:

CREATE OR REPLACE PROCEDURE DDSI_027.VERCUOTA
(id MONITOR.CODMONITOR%TYPE, 
	cursorUsuarios OUT SYS_REFCURSOR)
IS		
BEGIN
	OPEN cursorUsuarios FOR
		SELECT DISTINCT a.NOMBRE, a.PRECIOBASEMES,COUNT(s.NUMEROSOCIO) FROM
		(ACTIVIDAD a inner join REALIZA r on a.IDACTIVIDAD = r.IDACTIVIDAD) inner JOIN 
		SOCIO s on s.NUMEROSOCIO = r.NUMEROSOCIO
		WHERE a.MONITORRESPONSABLE = id
		GROUP BY a.NOMBRE, a.PRECIOBASEMES;
END VERCUOTA;
