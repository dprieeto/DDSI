--> Procedimiento en oracle:

CREATE OR REPLACE PROCEDURE DDSI_027.SOCIOSXACTIVIDAD
(id ACTIVIDAD.IDACTIVIDAD%TYPE, 
	cursorUsuarios OUT SYS_REFCURSOR)
IS
BEGIN
	OPEN cursorUsuarios FOR
		SELECT DISTINCT s.NOMBRE, s.CORREO FROM 
		REALIZA r inner join SOCIO s on s.NUMEROSOCIO=r.NUMEROSOCIO
		where r.IDACTIVIDAD = id;
END SOCIOSXACTIVIDAD;

CREATE DEFINER=`DDSI_027`@`%` PROCEDURE `DDSI_027`.`SOCIOSXACTIVIDAD`(in id Varchar(6))

		REALIZA r inner join SOCIO s on s.NUMEROSOCIO=r.NUMEROSOCIO
		where r.idActividad = id;
